80 def on_page_content(self, html: str, page: Page, **kwargs: Any) -> None: 

81 """Spell check everything. 

82 

83 Hook for the [`on_page_content` event](https://www.mkdocs.org/user-guide/plugins/#on_page_content). 

84 

85 Arguments: 

86 html: The HTML text. 

87 page: The page instance. 

88 **kwargs: Additional arguments passed by MkDocs. 

89 """ 

90 if page.file.src_path not in self.skip_files: 

91 words = get_words( 

92 html, 

93 known_words=self.known_words, 

94 min_length=self.min_length, 

95 max_capital=self.max_capital, 

96 ignore_code=self.ignore_code, 

97 allow_unicode=self.allow_unicode, 

98 ) 

99 for word in words: 

100 suggestions = self.spell.lookup(word, Verbosity.CLOSEST, max_edit_distance=2) 

101 if suggestions: 

102 candidates = "', '".join(suggestion.term for suggestion in suggestions if suggestion.term != word) 

103 if candidates: 

104 logger.warning(f"{page.file.src_path}: Misspelled '{word}', did you mean '{candidates}'?") 

105 else: 

106 logger.warning(f"{page.file.src_path}: Misspelled '{word}', no suggestions")
